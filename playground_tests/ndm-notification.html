ndm-notification
setSpeed 1000
open http://localhost:3000
waitForElementNotPresent css=span.loading__line
scrollIntoView text=ndm-notification
click text=Показать уведомление об успешном*
waitForVisible css=div.ndm-notification__body-msg--valigned
storeText css=div.ndm-notification__body-msg--valigned > span.ng-scope.ng-binding > span.ng-scope text
verifyEval "${text}"=="success!" true
click text=Показать информационное уведомление
waitForVisible css=div.ndm-notification__body-msg--valigned
storeText css=div.ndm-notification__body-msg--valigned text
verifyEval "${text}"=="info!" true
click text=Показать уведомление об ошибке
waitForVisible css=div.ndm-notification__body-msg--valigned
storeText css=div.ndm-notification__body-msg--valigned text
verifyEval "${text}"=="error!" true
